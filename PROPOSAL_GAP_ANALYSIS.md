model Campaign {
  id          String   @id @default(cuid())
  name        String
  workflow    Json     // Workflow definition
  status      String   // draft, active, paused, completed
  startDate   DateTime?
  endDate     DateTime?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
  
  leads       CampaignLead[]
  analytics   CampaignAnalytics[]
}

model CampaignLead {
  id          String   @id @default(cuid())
  campaignId  String
  personId    String?
  email       String
  status      String   // new, contacted, engaged, qualified, converted
  currentNode String?  // Current position in workflow
  
  campaign    Campaign @relation(fields: [campaignId], references: [id])
  person      Person?  @relation(fields: [personId], references: [id])
  touchpoints CampaignTouchpoint[]
}

model CampaignTouchpoint {
  id          String   @id @default(cuid())
  leadId      String
  type        String   // email, voicemail, linkedin, sms
  sentAt      DateTime
  opened      Boolean  @default(false)
  clicked     Boolean  @default(false)
  replied     Boolean  @default(false)
  
  lead        CampaignLead @relation(fields: [leadId], references: [id])
}

model CampaignAnalytics {
  id          String   @id @default(cuid())
  campaignId  String
  date        DateTime
  sent        Int      @default(0)
  opened      Int      @default(0)
  clicked     Int      @default(0)
  replied     Int      @default(0)
  qualified   Int      @default(0)
  converted   Int      @default(0)
  
  campaign    Campaign @relation(fields: [campaignId], references: [id])
}
